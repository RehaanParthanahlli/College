//=========================================================================
// SECTION X: ARRAYS IN JAVA
//=========================================================================
// Arrays are used to store multiple values of the SAME type in a single variable.
// They are reference data types (objects) in Java.
// Syntax: dataType[] arrayName = new dataType[size];

// Example: Declare and initialize
int[] numbers = new int[5];   // array of size 5 (default values = 0)
numbers[0] = 10;              // assign value at index 0
numbers[1] = 20;              // assign value at index 1

// Shortcut initialization
int[] marks = {85, 90, 75, 88, 92};

// Access elements using index (0-based)
System.out.println("First mark: " + marks[0]);   // 85
System.out.println("Last mark: " + marks[4]);    // 92

// Length property
System.out.println("Array length: " + marks.length);

//=========================================================================
// LOOPING THROUGH ARRAYS
//=========================================================================

// Using for loop
System.out.println("\n--- For Loop ---");
for (int i = 0; i < marks.length; i++) {
    System.out.println("marks[" + i + "] = " + marks[i]);
}

// Using enhanced for-each loop
System.out.println("\n--- For-Each Loop ---");
for (int m : marks) {
    System.out.println(m);
}

//=========================================================================
// MULTI-DIMENSIONAL ARRAYS
//=========================================================================
// Arrays of arrays (matrix-like structure)

int[][] matrix = {
    {1, 2, 3},
    {4, 5, 6},
    {7, 8, 9}
};

System.out.println("\n--- 2D Array ---");
System.out.println("matrix[0][0] = " + matrix[0][0]); // 1
System.out.println("matrix[2][2] = " + matrix[2][2]); // 9

// Loop through 2D array
for (int i = 0; i < matrix.length; i++) {
    for (int j = 0; j < matrix[i].length; j++) {
        System.out.print(matrix[i][j] + " ");
    }
    System.out.println();
}

//=========================================================================
// COMMON ARRAY OPERATIONS
//=========================================================================

// Sum of elements
int sum = 0;
for (int m : marks) {
    sum += m;
}
System.out.println("\nSum of marks: " + sum);

// Find maximum element
int max = marks[0];
for (int m : marks) {
    if (m > max) {
        max = m;
    }
}
System.out.println("Maximum mark: " + max);

//=========================================================================
// ARRAYS UTILITY CLASS (java.util.Arrays)
//=========================================================================

import java.util.Arrays;

System.out.println("\n--- Arrays Utility Methods ---");
System.out.println("Original: " + Arrays.toString(marks));

Arrays.sort(marks);   // Sort ascending
System.out.println("Sorted: " + Arrays.toString(marks));

int index = Arrays.binarySearch(marks, 90); // Search for 90
System.out.println("Index of 90: " + index);

int[] copy = Arrays.copyOf(marks, marks.length);
System.out.println("Copied array: " + Arrays.toString(copy));
